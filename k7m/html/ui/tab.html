// 기본실행
NETIVE.uiTab.init({ 
	id: 'id', 
	current: 0, 
	effect: 'eff-fade', 
	aria: true, 
	callback: function(v){console.log(v)} 
});
- effect : 'eff-fade', // 기본값 null : 전환효과 기본 show hide. 효과는css로 추가가능. 클래스명
- aria : true, // 기본값 false : true일때 aria관련 아이디 자동 생성
- reload : true // 기본값 true : true일때 탭클릭시 내용 변경 동장 취소

// 탭선택 필요 시 
NETIVE.uiTab.toggle({ id: 'id', current: 1, callback: function(v){console.log(v)} });
