// 기본실행
NETIVE.uiAcco.init({ id: 'exeAcco_1', current: 0, aria: true, autoclose: false, callback: function(v){console.log(v)} });
- aria : true // 기본값 false  : true일때 aria관련 아이디 자동 생성
- autoclose : false // 기본값 false  : true일때 선택된 아이템 외는 닫힘

// 선택 필요 시 
NETIVE.uiAcco.toggle({ id: 'exeAcco_1', current: 1, callback: function(v){console.log(v)} });

// 무조건 열기
NETIVE.uiAcco.toggle({ id: 'exeAcco_1', current: 1, state: 'open', callback: function(v){console.log(v)} });

// 무조건 닫기
NETIVE.uiAcco.toggle({ id: 'exeAcco_1', current: 1, state: 'close', callback: function(v){console.log(v)} });
