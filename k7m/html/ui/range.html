NETIVE.uiRange.init({
	id : "uiRange",
	step : 10,            // 단계설정
	min : 1000,           // 최소값
	max : 6000,           // 최대값
	now : [1500, 2500],   // 설정값, 단계설정으로 맞추어진 근사값으로 변경됨.
	tooltip : true,       // 툴팁여부
	unit: '만원',         // 단위 텍스트
	txt_s : '이하',       // 첫번째 값 추가 텍스트
	txt_e : '이상',       // 마지막 값 추가 텍스트
	acc : true,           // 접근성 select 박스연결
	callback: callback    // callback 함수
});
function callback(v){
	console.log(v.min, v.max);
}
