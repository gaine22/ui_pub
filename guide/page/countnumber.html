var data = {
	'selector': '.exe_count1',
	
	// true 콤마 자동생성필요, false 콤마 자동생성불필요
	'comma': true,
	
	// 문자식일경우'123,23.00', 값이 없을경우 false           
	'value': 54231258.135,
	
	// false일 경우 기본값 1000
	'speed': 1000,
	
	// false일 경우 기본값 'easeInOutQuad'      
	'eff': 'easeInOutQuad'        
};
$._uiCountNumber.start(data);
